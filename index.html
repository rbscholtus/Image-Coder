// make image greyscale
for (int x = 0; x < width(); x++) {
	setProgress(x, width());
	for (int y = 0; y < height(); y++) {
		int avg = ( red(x,y) + green(x,y) + blue(x,y) ) / 3;
		setRGB(x, y, makeRGB(avg, avg, avg));
	}
}

// blur image
for (int x = 1; x < width()-1; x++) {
	setProgress(x-1, width()-2);
	for (int y = 1; y < height()-1; y++) {
		int red = (red(x,y-1) + red(x+1,y) + red(x,y+1) + red(x-1,y) ) / 4;
		int green = (green(x,y-1) + green(x+1,y) + green(x,y+1) + green(x-1,y) ) / 4;
		int blue = (blue(x,y-1) + blue(x+1,y) + blue(x,y+1) + blue(x-1,y) ) / 4;
		setRGB(x, y, makeRGB(red, green, blue));
	}
}

// flip image horizontally
for (int x = 0; x < width(); x++) {
	setProgress(x, width());
	for (int y = 0; y < height()/2; y++) {
		int p = getRGB(x, y);
		setRGB(x, y, getRGB(x,height()-y-1));
		setRGB(x, height()-y-1, p);
	}
}

// edge detection
for (int y = 0; y < height(); y++) {
	setProgress(y, height());
    for (int x = 0; x < width(); x++) {
		int avg1 = ( red(x,y) + green(x,y) + blue(x,y) ) / 3;
		int avg2 = ( red(x+1,y) + green(x+1,y) + blue(x+1,y) ) / 3;
		int avg3 = ( red(x,y+1) + green(x,y+1) + blue(x,y+1) ) / 3;
		int diff = Math.abs(avg1-avg2) + Math.abs(avg1-avg3);
		setRGB(x, y, makeRGB(diff, diff, diff));
	}
}

for (int x = 0; x<getWidth()-1; x++) {
	setProgress(x, getWidth()-1);
	for (int y = 0; y<getHeight(); y++) {
		int avg = ( getR(x,y) + getG(x,y) + getB(x,y) ) / 3;
		int avg2 = ( getR(x+1,y) + getG(x+1,y) + getB(x+1,y) ) / 3;
